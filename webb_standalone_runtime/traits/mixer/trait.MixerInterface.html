    ) -> Result<TreeId, DispatchError>;

        leaf: Element
    ) -> Result<(), DispatchError>;

        id: TreeId,

        fee: Balance,
        refund: Balance
    ) -> Result<(), DispatchError>;

    ) -> Result<(), DispatchError>;
}
